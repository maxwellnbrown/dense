164  void calc_neighbor_2d() {
165  for (dense::Natural i = 0; i < _cells_total; ++i) {
166  bool is_former_edge = i % _width_total == 0;
167  bool is_latter_edge = (i + 1) % _width_total == 0;
168  bool is_even = i % 2 == 0;
169  auto la = (is_former_edge || !is_even) ? _width_total - 1 : -1;
170  auto ra = !(is_latter_edge || is_even) ? _width_total + 1 : 1;
171 
172  auto top = (i - _width_total + _cells_total) % _cells_total;
173  auto bottom = (i + _width_total ) % _cells_total;
174  auto bottom_right = (i + ra ) % _cells_total;
175  auto top_left = (i + la ) % _cells_total;
176  auto top_right = (i - _width_total + ra + _cells_total) % _cells_total;
177  auto bottom_left = (i - _width_total + la + _cells_total) % _cells_total;
178 
179  if (is_former_edge) {
180  _neighbors[i] = { top, top_right, top_left, bottom_left };
181  _numNeighbors[i] = 4;
182  } else if (is_latter_edge) {
183  _neighbors[i] = { top, top_right, bottom_right, bottom };
184  _numNeighbors[i] = 4;
185  } else {
186  _neighbors[i] = { top, top_right, bottom_right, bottom, top_left, bottom_left };
187  _numNeighbors[i] = 6;
188  }
189  }
